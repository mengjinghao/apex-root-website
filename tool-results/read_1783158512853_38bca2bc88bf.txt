    78→    // ── Hide / Spoof ──
    79→
    80→    /**
    81→     * 隐藏策略切换：UNIDIRECTIONAL / FULL / TARGETED
    82→     * 接线：调用 NativeBridge.enableHideMode(appUid) 真正激活 native 层隐藏。
    83→     * UNIDIRECTIONAL → 启用 Hide 模式
    84→     * FULL → 启用 Game 模式（aggressive 隐藏）
    85→     * TARGETED → 关闭 native 隐藏（仅依赖检测层）
    86→     */
    87→    fun updateHideStrategy(strategy: HideStrategy) {
    88→        _settings.value = _settings.value.copy(hideStrategy = strategy); persist()
    89→        // 接线 native 层
    90→        viewModelScope.launch(Dispatchers.IO) {
    91→            val appUid = getApplication<Application>().applicationInfo.uid
    92→            when (strategy) {
    93→                HideStrategy.UNIDIRECTIONAL -> { NativeBridge.enableHideMode(appUid) }
    94→                HideStrategy.FULL -> { NativeBridge.enableGameMode(appUid) }
    95→                HideStrategy.TARGETED -> { NativeBridge.disableHideMode() }
    96→            }
    97→        }
    98→    }

   427→    }
