   735→ * 顶部「运行中」状态条。
   736→ *
   737→ * 高斯玻璃（liquidGlass）+ 圆角的精致状态条，置于主界面顶部：
   738→ *  - 左侧：脉动绿点 + "运行中" 文本 + 引擎就绪状态
   739→ *  - 右侧：版本号 + 点击进入权限中心
   740→ */
   741→@Composable
   742→private fun RunningStatusBar(
   743→    nativeAvailable: Boolean,
   744→    onNavigateToPermissions: (() -> Unit)?
   745→) {
   746→    val context = androidx.compose.ui.platform.LocalContext.current
   747→    val isDark = LocalIsDarkTheme.current
   748→
   749→    // 读取版本号
   750→    val versionName = remember {
   751→        try {
   752→            val pm = context.packageManager
   753→            pm.getPackageInfo(context.packageName, 0).versionName ?: "1.0.3"
   754→        } catch (e: Throwable) {
   755→            "1.0.3"
   756→        }
   757→    }
   758→
   759→    // 脉动动画
   760→    val transition = rememberInfiniteTransition(label = "runStatus")
   761→    val pulseAlpha by transition.animateFloat(
   762→        initialValue = 0.4f,
   763→        targetValue = 1f,
   764→        animationSpec = infiniteRepeatable(
   765→            animation = tween(900, easing = androidx.compose.animation.core.LinearEasing),
   766→            repeatMode = androidx.compose.animation.core.RepeatMode.Reverse
   767→        ),
   768→        label = "pulse"
   769→    )
   770→    val pulseScale by transition.animateFloat(
   771→        initialValue = 0.85f,
   772→        targetValue = 1.15f,
   773→        animationSpec = infiniteRepeatable(
   774→            animation = tween(900, easing = androidx.compose.animation.core.LinearEasing),
   775→            repeatMode = androidx.compose.animation.core.RepeatMode.Reverse
   776→        ),
   777→        label = "pulseScale"
   778→    )
   779→
   780→    Box(
   781→        modifier = Modifier
   782→            .fillMaxWidth()
   783→            .padding(horizontal = 16.dp)
   784→            .height(52.dp)
   785→            .clip(RoundedCornerShape(18.dp))
   786→            .liquidGlass(
   787→                cornerRadius = 18.dp,
   788→                baseColor = if (isDark) Color.White.copy(alpha = 0.06f) else Color.Black.copy(alpha = 0.025f)
   789→            )
   790→            .clickable(
   791→                interactionSource = remember { androidx.compose.foundation.interaction.MutableInteractionSource() },
   792→                indication = null
   793→            ) { onNavigateToPermissions?.invoke() }
   794→            .padding(horizontal = 16.dp),
   795→        contentAlignment = Alignment.CenterStart
   796→    ) {
   797→        Row(
   798→            modifier = Modifier.fillMaxWidth(),
   799→            verticalAlignment = Alignment.CenterVertically
   800→        ) {
   801→            // 脉动状态点
   802→            Box(
   803→                modifier = Modifier.size((10 * pulseScale).dp),
   804→                contentAlignment = Alignment.Center
   805→            ) {
   806→                Box(
   807→                    Modifier
   808→                        .size(8.dp)
   809→                        .background(AccentMint.copy(alpha = pulseAlpha), CircleShape)
   810→                )
   811→            }
   812→            Spacer(Modifier.width(10.dp))
   813→            // 运行中
   814→            Text(
   815→                "运行中",
   816→                fontSize = 14.sp,
   817→                fontWeight = FontWeight.SemiBold,
   818→                color = TextPrimary
   819→            )
   820→            Spacer(Modifier.width(8.dp))
   821→            // 引擎状态
   822→            Box(
   823→                Modifier
   824→                    .clip(RoundedCornerShape(6.dp))
   825→                    .background(
   826→                        if (nativeAvailable) AccentMint.copy(alpha = 0.14f) else AccentGold.copy(alpha = 0.14f)
   827→                    )
   828→                    .padding(horizontal = 7.dp, vertical = 2.dp)
   829→            ) {
   830→                Text(

   837→            Spacer(Modifier.weight(1f))
   838→            // 版本号
   839→            Row(verticalAlignment = Alignment.CenterVertically) {
   840→                Text(
   841→                    "v$versionName",
   842→                    fontSize = 12.sp,
   843→                    color = TextTertiary,
