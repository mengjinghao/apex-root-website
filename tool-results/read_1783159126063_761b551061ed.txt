   361→@Composable
   362→private fun DetectionLevelSelector(current: DetectionLevel, onSelect: (DetectionLevel) -> Unit) {
   363→    // 优化：从 FilterChip 改为 Slider 滑动开关，符合用户要求
   364→    // 4 个级别：QUICK(0) / STANDARD(1) / DEEP(2) / FORENSIC(3)
   365→    // steps = 2 表示在 0..3 之间有 2 个内部停靠点（即 1 和 2），共 4 个位置
   366→    Column(
   367→        modifier = Modifier.fillMaxWidth().padding(horizontal = 20.dp, vertical = 8.dp)
   368→    ) {
   369→        // 当前级别显示
   370→        Row(
   371→            modifier = Modifier.fillMaxWidth(),
   372→            horizontalArrangement = Arrangement.SpaceBetween,
   373→            verticalAlignment = Alignment.CenterVertically
   374→        ) {
   375→            Text(
   376→                current.label,
   377→                fontSize = 14.sp,
   378→                fontWeight = FontWeight.SemiBold,
   379→                color = AccentPurple
   380→            )
   381→            Text(
   382→                "级别 ${current.value + 1} / ${DetectionLevel.entries.size}",
   383→                fontSize = 11.sp,
   384→                color = TextTertiary
   385→            )
   386→        }
   387→        Spacer(Modifier.height(4.dp))
   388→        // Slider 主体
   389→        Row(
   390→            modifier = Modifier.fillMaxWidth(),
   391→            verticalAlignment = Alignment.CenterVertically
   392→        ) {
   393→            Text("快速", fontSize = 10.sp, color = TextTertiary)
   394→            Slider(
   395→                value = current.value.toFloat(),
   396→                onValueChange = { value ->
   397→                    val newIndex = value.toInt()
   398→                    DetectionLevel.fromValue(newIndex)?.let { onSelect(it) }
   399→                },
   400→                valueRange = 0f..(DetectionLevel.entries.size - 1).toFloat(),
   401→                steps = DetectionLevel.entries.size - 2,  // N 个停靠点需要 N-2 个 steps
   402→                modifier = Modifier.weight(1f).padding(horizontal = 8.dp),
   403→                colors = SliderDefaults.colors(
   404→                    thumbColor = AccentPurple,
   405→                    activeTrackColor = AccentPurple,
   406→                    inactiveTrackColor = AccentPurple.copy(alpha = 0.15f)
   407→                )
   408→            )
   409→            Text("取证", fontSize = 10.sp, color = TextTertiary)
   410→        }
   411→        // 级别标签
   412→        Row(
   413→            modifier = Modifier.fillMaxWidth().padding(horizontal = 4.dp),
   414→            horizontalArrangement = Arrangement.SpaceBetween
   415→        ) {
   416→            DetectionLevel.entries.forEach { level ->
   417→                Text(
   418→                    level.name.first().toString(),
   419→                    fontSize = 9.sp,
   420→                    color = if (level == current) AccentPurple else TextTertiary
   421→                )
   422→            }
   423→        }
   424→    }
   425→}

   472→private fun PluginToggles(settings: AppSettings, vm: SettingsViewModel) {
   473→    val plugins = listOf(
   474→        1 to "L1 属性裸读", 2 to "L2 ART 注入", 3 to "L3 进程/内存",

   477→    )
   478→    Column(modifier = Modifier.padding(horizontal = 20.dp)) {
   479→        plugins.forEach { (index, name) ->
   480→            val enabled = when (index) {
   481→                1 -> settings.pluginL1Enabled; 2 -> settings.pluginL2Enabled
   482→                3 -> settings.pluginL3Enabled; 4 -> settings.pluginL4Enabled
   483→                5 -> settings.pluginL5Enabled; 6 -> settings.pluginL6Enabled
   484→                7 -> settings.pluginL7Enabled; else -> true
   485→            }
   486→            Row(
   487→                modifier = Modifier.fillMaxWidth().padding(vertical = 4.dp),
   488→                verticalAlignment = Alignment.CenterVertically
   489→            ) {
   490→                Text(name, fontSize = 12.sp, color = TextSecondary, modifier = Modifier.weight(1f))
   491→                Switch(
   492→                    checked = enabled,
   493→                    onCheckedChange = { vm.updatePluginEnabled(index, it) },
   494→                    colors = SwitchDefaults.colors(
   495→                        checkedThumbColor = AccentPurple,
   496→                        checkedTrackColor = AccentPurple.copy(alpha = 0.25f),
   497→                        uncheckedThumbColor = TextTertiary,
   498→                        uncheckedTrackColor = TextTertiary.copy(alpha = 0.15f)

   500→                )
   501→            }
   502→        }
