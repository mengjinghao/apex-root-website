   517→    fun getRecommendation(layer: String): FixRecommendation? {
   518→        // Sort keys by length descending to match most specific first
   519→        // This prevents "固件" matching before "固件完整性"
   520→        val sortedKeys = recommendations.keys.sortedByDescending { it.length }
   521→        val matchedKey = sortedKeys.firstOrNull { key ->
   522→            when (key) {
   523→                "系统调用时序" -> layer == "系统调用时序" || layer == "系统调用"
   524→                "APatch KPM" -> layer.contains("APatch") && layer.contains("KPM")
   525→                "固件完整性" -> layer.contains("固件") && (layer.contains("完整") || layer.contains("分区"))
   526→                else -> layer == key || (layer.contains(key) && key.length >= 2)
   527→            }
   528→        }
   529→        return matchedKey?.let { recommendations[it] }
   530→    }
   531→
   532→    fun getAllRecommendations(): Collection<FixRecommendation> {
   533→        return recommendations.values
   534→    }
   535→
   536→    fun getRecommendationsForLayers(layers: List<String>): List<FixRecommendation> {
   537→        return layers.mapNotNull { getRecommendation(it) }
   538→            .sortedByDescending { it.priority }
   539→            .distinctBy { it.id }
   540→    }
